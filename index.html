整理文件

from PIL import Image #入PIL模块用于读取图片，也可使用opencv
import os

if __name__ == "__main__":
    path = "C:/Users/jyh/Desktop/new/panorama/images/loadingscreens"    # 修改载入的文件夹
    path_dir = os.path.join(os.getcwd(),"new")                          # 创建合并后的文件夹
    i = 0
    folder = os.path.exists(path_dir)
    if not folder:
        os.makedirs(path_dir)
    # 循环遍历lfw数据集下的所有子文件
    for name_file in os.listdir(path):
    # 遍历子文件下的所有图片文件
        for img_file in os.listdir(path + "/" + name_file):
            # print(img_file)    # 打印当前读取的图片名
            img = Image.open(path+"/"+name_file+"/"+img_file)           
            save_path = str(i) + ".png"
            i = i + 1
            Image.Image.save(img,path_dir+"/"+ save_path)
